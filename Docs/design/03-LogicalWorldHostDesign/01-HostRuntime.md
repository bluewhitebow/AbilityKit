## 2. HostRuntime 接口

```csharp
public interface IHostRuntime
{
    // ========== World 管理 ==========

    /// <summary>获取 World 管理器</summary>
    IWorldManager Worlds { get; }

    /// <summary>创建世界</summary>
    IWorld CreateWorld(WorldCreateOptions options);

    /// <summary>销毁世界</summary>
    bool DestroyWorld(WorldId id);

    // ========== 网络连接 ==========

    /// <summary>连接</summary>
    void Connect(IServerConnection connection);

    /// <summary>断开连接</summary>
    void Disconnect(ServerClientId clientId);

    /// <summary>广播消息</summary>
    void Broadcast(ServerMessage message);

    // ========== 生命周期 ==========

    /// <summary>Tick</summary>
    void Tick(float deltaTime);

    /// <summary>启动</summary>
    void Start();

    /// <summary>停止</summary>
    void Stop();
}
```
